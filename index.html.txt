/* ================= BASE DE DADOS ================= */
const LS_KEY='banhos';
const uid = () => Date.now().toString(36)+Math.random().toString(36).slice(2,8);
const loadBanhos=()=>{try{return JSON.parse(localStorage.getItem(LS_KEY))||[]}catch{return[]}};
const saveBanhos=(arr)=>localStorage.setItem(LS_KEY,JSON.stringify(arr));

/* ================= AGENDA & FINANCEIRO ================= */

  if(confirm('Deseja realmente limpar todo o hist√≥rico de banhos?')){
    localStorage.removeItem(LS_KEY);
    renderCalendario();
    renderFinanceiro();
  }
}

/* ================= CALEND√ÅRIO ================= */
const nomesMeses=["janeiro","fevereiro","mar√ßo","abril","maio","junho","julho","agosto","setembro","outubro","novembro","dezembro"];
const nomesDias=["Seg","Ter","Qua","Qui","Sex","S√°b","Dom"];
let hoje = new Date();

function renderCalendario(){
  const calendario = document.getElementById('calendario');
  calendario.innerHTML='';
  const banhos = loadBanhos();

  for(let m=0;m<=1;m++){
    const mes = hoje.getMonth()+m;
    const ano = hoje.getFullYear();
    const primeiroDia = new Date(ano,mes,1);
    const ultimoDia = new Date(ano, mes+1,0);
    const diasMes = ultimoDia.getDate();
    const primeiroDiaSemana = (primeiroDia.getDay()+6)%7;

    const mesDiv = document.createElement('div');
    mesDiv.classList.add('mes');
    mesDiv.innerHTML=`<h2>${nomesMeses[mes]} ${ano}</h2>
      <div class="dias-semana">${nomesDias.map(d=>`<div>${d}</div>`).join('')}</div>
      <div class="dias"></div>`;
    const diasGrid = mesDiv.querySelector('.dias');

    for(let i=0;i<primeiroDiaSemana;i++){diasGrid.innerHTML+='<div class="dia"></div>'}

    for(let dia=1;dia<=diasMes;dia++){
      const dataObj = new Date(ano, mes, dia);
      const diaDiv = document.createElement('div');
      diaDiv.classList.add('dia');
      if(dia===hoje.getDate() && mes===hoje.getMonth() && ano===hoje.getFullYear()) diaDiv.classList.add('hoje');
      diaDiv.innerHTML=`<span class="dia-num">${dia}</span>`;

      const banhosDia = banhos.filter(b=>{
        const partes = b.data.split('-'); 
        const bData = new Date(partes[0], partes[1]-1, partes[2]);
        return bData.getDate()===dia && bData.getMonth()===mes && bData.getFullYear()===ano && b.status!=='cancelado_sistema';

      });
      banhosDia.forEach(b=>{
        const span = document.createElement('button');
        span.classList.add('badge',b.tipo);
        if(b.status) span.classList.add(b.status);
        span.textContent = b.tipo==='pacote'?`${b.cliente} [${b.seq}/4]`:`${b.cliente}`;
        span.onclick=()=>abrirModal(b);
        diaDiv.appendChild(span);
      });

      diasGrid.appendChild(diaDiv);
    }

    calendario.appendChild(mesDiv);
  }
}

// ===== Adicionar Banho (Corrigido) =====
function adicionarBanho(){
  const tipo = document.getElementById('tipoBanho').value;
  const tutor = document.getElementById('tutor').value.trim();
  const cliente = document.getElementById('cliente').value.trim();
  const data = document.getElementById('dataBanho').value;
  const hora = document.getElementById('horario').value;
  const valor = parseFloat(document.getElementById('valor').value) || 0;

  if(!tutor || !cliente || !data){
    alert('Preencha todos os campos!');
    return;
  }

  const banhos = loadBanhos();
  const idBase = uid();

  const partes = data.split('-');
  const dataObj = new Date(partes[0], partes[1]-1, partes[2]);

  // Fun√ß√£o para formatar data local em YYYY-MM-DD
  const formatLocalDate = (d) => {
    return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
  };

  if(tipo==='avulso'){
    banhos.push({
      id: idBase,
      tipo,
      tutor,
      cliente,
      data: formatLocalDate(dataObj),
      hora,
      valor,
      status:'ativo'
    });
  } else if(tipo==='pacote'){
    for(let i=0;i<4;i++){
      const novaData = new Date(dataObj.getTime());
      novaData.setDate(novaData.getDate() + i*7);
      novaData.setHours(0,0,0,0); // garante o dia correto
      const valorSemana = +(valor/4).toFixed(2);
      banhos.push({
        id: idBase+'-'+i,
        tipo,
        tutor,
        cliente,
        data: formatLocalDate(novaData),
        hora,
        valor: valorSemana,
        status:'ativo',
        seq: i+1
      });
    }
  }

  saveBanhos(banhos);
  renderCalendario();
  renderFinanceiro();

  document.getElementById('tutor').value='';
  document.getElementById('cliente').value='';
  document.getElementById('dataBanho').value='';
  document.getElementById('horario').value='';
  document.getElementById('valor').value='';
}


function renderFinanceiro(){
  const banhos = loadBanhos();
  const hojeStr = hoje.toISOString().slice(0,10);
  const totalDia = banhos.filter(b=>b.data===hojeStr && b.status==='ativo').reduce((a,b)=>a+b.valor,0);
  const totalMes = banhos.filter(b=>{const d=new Date(b.data); return d.getMonth()===hoje.getMonth() && d.getFullYear()===hoje.getFullYear() && b.status==='ativo'}).reduce((a,b)=>a+b.valor,0);
  const ticketMedio = banhos.filter(b=>b.status==='ativo').length? (banhos.filter(b=>b.status==='ativo').reduce((a,b)=>a+b.valor,0)/banhos.filter(b=>b.status==='ativo').length) : 0;
  const totalClientes = [...new Set(banhos.filter(b=>b.status==='ativo').map(b=>b.cliente))].length;

  document.getElementById('totalDia').textContent=`R$ ${totalDia.toFixed(2)}`;
  document.getElementById('totalMes').textContent=`R$ ${totalMes.toFixed(2)}`;
  document.getElementById('ticketMedio').textContent=`R$ ${ticketMedio.toFixed(2)}`;
  document.getElementById('totalClientes').textContent=totalClientes;

  const tbody = document.getElementById('listaBanhos');
  tbody.innerHTML='';
  banhos.forEach(b=>{
    const tr = document.createElement('tr');
    tr.innerHTML=`<td>${b.id}</td>
      <td>${b.cliente}</td>
      <td>${b.tutor}</td>
      <td>${b.data}</td>
      <td>${b.hora}</td>
      <td>R$ ${b.valor.toFixed(2)}</td>
      <td>${b.tipo}</td>
      <td>${b.status}</td>
      <td class="t-actions">
        <button class="mini edit" onclick="abrirModal('${b.id}')">‚úèÔ∏è</button>
        <button class="mini" onclick="cancelarModalTabela('${b.id}','cliente')">üö´</button>
      </td>`;
    tbody.appendChild(tr);
  });
}

// Modal e cancelamento simplificados
let modalAtual=null;

function fecharModal(e){ if(!e || e.target===document.getElementById('modalBackdrop')){ document.getElementById('modalBackdrop').style.display='none'; modalAtual=null; } }
function salvarEdicao(){ if(!modalAtual)return; const banhos=loadBanhos(); const b = banhos.find(x=>x.id===modalAtual.id); b.data=document.getElementById('modalData').value; b.hora=document.getElementById('modalHora').value; b.valor=parseFloat(document.getElementById('modalValor').value)||0; saveBanhos(banhos); fecharModal(); renderCalendario(); renderFinanceiro();}
function cancelarModal(tipo){ if(!modalAtual)return; const banhos=loadBanhos(); const b=banhos.find(x=>x.id===modalAtual.id); b.status=tipo==='cliente'?'cancelado_cliente':'cancelado_sistema'; saveBanhos(banhos); fecharModal(); renderCalendario(); renderFinanceiro();}
function cancelarModalTabela(id,tipo){ modalAtual = loadBanhos().find(x=>x.id===id); cancelarModal(tipo);}
function reativarModal(){ if(!modalAtual)return; const banhos=loadBanhos(); const b=banhos.find(x=>x.id===modalAtual.id); b.status='ativo'; saveBanhos(banhos); fecharModal(); renderCalendario(); renderFinanceiro();}
